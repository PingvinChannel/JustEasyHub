getgenv().part = Instance.new('Part')
local VirtualUser = game:GetService("VirtualUser")
game:GetService("Players").LocalPlayer.Idled:connect(
function()
VirtualUser:CaptureController()
VirtualUser:ClickButton2(Vector2.new())
end)
part.Anchored = true
part.Size = Vector3.new(100000000,5,100000000)
part.CFrame = CFrame.new(0,10000,0)
part.Parent = game.Workspace
game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = part.CFrame * CFrame.new(0,10,0)
getgenv().autofarm = true
local driving = true
while getgenv().autofarm do
   for a,b in next, game:GetService('Workspace')['$cars']:children() do
       if pcall(function() return b.State end) and pcall(function() return b.State.Driver end) then
           if tostring(b.State.Driver.Value) == tostring(game.Players.LocalPlayer.Name) then
               part.Position = Vector3.new(b.RootPart.Position.X, 10000, b.RootPart.Position.Z)
                if not driving then
                    driving = true
                    game:GetService'VirtualUser':SetKeyUp("w")
                    game:GetService'VirtualUser':SetKeyDown("w")
                end
               break
           end
       end
   end
   wait(.2)
end
